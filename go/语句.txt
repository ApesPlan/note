func eval(a, b int, op string) int {
  var result int
  switch op {
  case "+":
    result = a + b
  case "-":
    result = a - b
  default:
    panic("adfadf" + op)
  }
  return result
}

func grade(score int) string {
  switch { // switch后可以没有表达式
  case score < 60:
    return "F"
  case score < 80:
    return "C"
  default:
    return "A"
  }
}
